    ```bash
    cd ~/il_ws/src/lerobot && lerobot-train \
      --dataset.repo_id=${HF_USER}/pick \
      --policy.type=act \
      --output_dir=outputs/train/omx_act_policy2 \
      --job_name=act_record-test2 \
      --policy.device=cuda \
      --wandb.enable=false \
      --policy.repo_id=${HF_USER}/omx_act_policy2 \
      --batch_size=16 \
      --save_checkpoint=true \
      --save_freq=10000 \
      --steps=100000
    ```
